有限元作业——Mathematica

2020-12-02 mma

题为计算长直接地金属槽内电场，为强迫边界条件。

计算时采用了所谓的“划0置1”法。课程讲义中通过调整节点编号的方式来避免这种方法带来的矩阵处理上的复杂，但增加了节点划分的任务量。此处采用讲义中提到的直线内插法划分节点，而Mathematica强大的功能也使“划0置1”法的实现变得十分简单。

ClearAll["Global`*"]
(*节点坐标*)
x = Flatten[(Range[7]^0*#/10) & /@ (Range[6] - 1)];
y = Flatten[((Range[7] - 1)/10) & /@ Range[6]];
(*节点编号*)
ie = Join[Flatten[(7 (# - 1) + Range[6]) & /@ Range[5]], 
   Flatten[(7 (# - 1) + Range[6]) & /@ Range[5]]];
je = Join[ie[[Range[30]]] + 8, ie[[Range[30] + 30]] + 7];
me = Join[ie[[Range[30]]] + 1, ie[[Range[30] + 30]] + 8];
(*强加边界条件*)
d = {Join[Table[0, 12], Table[100, 5]], 
   Join[Range[7], 7*Range[5] + 1, 7*Range[5] + 7]};
(*所求节点编号*)
a = Flatten[(Range[5] + 7 # + 1) & /@ Range[5]];
(*单元分析*)
K = Table[0, {i, 1, 42}, {j, 1, 42}];
For[t = 1, t <= Length[ie], t++,
  b[ie[[t]]] = y[[je[[t]]]] - y[[me[[t]]]];
  b[je[[t]]] = y[[me[[t]]]] - y[[ie[[t]]]];
  b[me[[t]]] = y[[ie[[t]]]] - y[[je[[t]]]];
  c[ie[[t]]] = x[[me[[t]]]] - x[[je[[t]]]];
  c[je[[t]]] = x[[ie[[t]]]] - x[[me[[t]]]];
  c[me[[t]]] = x[[je[[t]]]] - x[[ie[[t]]]];
  sideta[t] = 2*(b[ie[[t]]]*c[je[[t]]] - b[je[[t]]]*c[ie[[t]]]);
  ff = {ie[[t]], je[[t]], me[[t]]};
  For[f1 = 1, f1 <= 3, f1++,
   p = ff[[f1]];
   For[f2 = f1, f2 <= 3, f2++,
    q = ff[[f2]];
    K[[p, q]] = K[[p, q]] + (b[p]*b[q] + c[p]*c[q])/sideta[t];
    K[[q, p]] = K[[p, q]];
    ];
   ];
  ];
(*求解线性方程组*)
R = Table[0, 42];
R[[d[[2]]]] = d[[1]];
Do[R[[i]] = -Plus @@ (K[[i, #]] & /@ d[[2]]*d[[1]]), {i, a}]
Do[Part[K, All, i] = 0, {i, d[[2]]}];
Do[Part[K, i, All] = 0, {i, d[[2]]}];
Do[K[[i, i]] = 1, {i, d[[2]]}];
A = N[LinearSolve[K, R]];
(*结果输出*)
A = Join[A[[Range[7] + 7 # - 7]] & /@ Range[6], 
   A[[Range[7] + 42 - 7 #]] & /@ (Range[5] + 1)];
ListPlot3D[Transpose[A], ColorFunction -> "Rainbow"]
