差分法作业——Mathematica

2020-10-24 mma

题目

T形电机电枢槽气隙磁位分布，忽略曲度效应。

差分法，Mathematica 12.0 代码：


(*基本参数*)
bm = 74; bn = 11; bab = 8; bbc = 6; 
mul = 1;(*网格倍率*)
m = mul*bm; n = mul*bn; ab = mul*bab; bc = mul*bbc;
alpha = (2 - Pi*Sqrt[2 (m^2 + n^2)]/(m n));
(*初始化*)
A = Table[a[i, j], {i, m + 1}, {j, n + 1}];
For[j = 1, j <= n + 1, j++, a[1, j] = 100];
For[j = 1, j <= bc + 1, j++, a[ab + 1, j] = 0];
For[i = ab + 2, i <= m + 1, i++, a[i, bc + 1] = 0];
For[j = bc + 2, j <= n + 1, j++, a[m + 1, j] = 0];
For[i = 2, i < ab + 1, i++,
  For[j = 1, j <= bc + 1, j++,
    a[i, j] = 100.0 - 100.0/ab*(i - 1)];
  ];
For[i = 2, i < m + 1, i++,
  For[j = bc + 2, j <= n + 1, j++,
    a[i, j] = 100.0 - 100.0/m*(i - 1)];
  ];
For[i = ab + 2, i <= m + 1, i++,
  For[j = 1, j < bc + 1, j++, a[i, j] = 0];
  ];
(*A = Table[a[i,j],{i,m+1},{j,n+1}];
A//Grid*)
(*迭代*)
flag = True;
While[flag,
  flagn = 0;
  For[i = 2, i <= ab, i++,
   For[j = 1, j <= bc + 1, j++,
     Al = a[i, j];
     If[j == 1, 
      a[i, j] = 
       N[a[i, j] + 
         alpha/4*(a[i - 1, j] + 2 a[i, j + 1] + a[i + 1, j] - 
            4 a[i, j])],
      a[i, j] = 
        N[a[i, j] + 
          alpha/4*(a[i, j - 1] + a[i - 1, j] + a[i, j + 1] + 
             a[i + 1, j] - 4 a[i, j])];
      ];
     If[Al != 0, If[Abs[(Al - a[i, j])/Al] > 0.001, flagn++, Null], 
      If[Abs[a[i, j]] > 0.001, flagn++, Null]];
     ];
   ];
  For[i = 2, i <= m, i++,
   For[j = bc + 2, j <= n + 1, j++,
     Al = a[i, j];
     If[j == n + 1, 
      a[i, j] = 
       N[a[i, j] + 
         alpha/4*(a[i - 1, j] + 2 a[i, j - 1] + a[i + 1, j] - 
            4 a[i, j])],
      a[i, j] = 
        N[a[i, j] + 
          alpha/4*(a[i, j - 1] + a[i - 1, j] + a[i, j + 1] + 
             a[i + 1, j] - 4 a[i, j])];
      ];
     If[Al != 0, If[Abs[(Al - a[i, j])/Al] > 0.001, flagn++, Null], 
      If[Abs[a[i, j]] > 0.001, flagn++, Null]];
     ];
   ];
  If[flagn != 0, flag = True, flag = False];
  ];
(*输出*)
A = Table[a[i, j], {i, m + 1}, {j, n + 1}];
Export["C:\\Users\\Linber\\Desktop\\sim cixi.xlsx", A];
A = Transpose[A];
ListPlot3D[A, PlotRange -> All, ColorFunction -> "Rainbow"]
