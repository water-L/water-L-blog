void LS(float K[12][12], float d[6])
{

    float temp;
    float R[6] = {0}, A[6] = {0};
    float *matrix[7];

        matrix[i]= K[i];
    }
    
    for (i = 0; i < 6; i++){
        for (j = 6; j < 12; j++){
            R[i] += -K[i][j] * d[j - 6];
        }

void gauss_elimination(float *matrix[7])
{
    int i, j, k;
    float coe;

    for(i=0;i<6;i++){
        exchange(matrix, i);
        for(j=i+1;j<6;j++){
            coe = *(matrix[i]+j) / *(matrix[i]+i);
            for(k=i;k<7;k++){
                *(matrix[k]+j) -= coe * *(matrix[k]+i);
            }           
        }
    }
    
    return;

}

void exchange(float *matrix[7], int flag)
{
    int i;
    float*temp;

    for(i=flag+1;i<6;i++){
        if(fabs(*(matrix[flag]+flag))<fabs(*(matrix[flag]+i))){
            temp = matrix[flag];
            matrix[flag] = matrix[i];
            matrix[i] = temp;
        }
    }
    
    return;

}
