有限元作业——C语言

2020-12-07 #C

好久没用C语言写东西了，上次还是年初的时候写了个生日祝福的小程序……

对比mma和C，果然还是站在别人的基础上方便（ノへ￣、）

言归正传，这是一个很简单的FEM的题目，唯一难的地方是要用C语言写。

代码如下：

#include <math.h>

#include <stdio.h>

#include <stdlib.h>

/*求解线性方程组*/
void LS(float K[12][12], float d[6]);
void gauss_elimination(float *[7]);
void exchange(float *[7], int flag);

int main()
{
    /*网格坐标及节点编号*/
    float x[12] = {3.0, 6.0, 3.0, 6.0, 3.0, 6.0, 0.0, 0.0, 0.0, 9.0, 9.0, 9.0};
    float y[12] = {0.0, 0.0, 3.0, 3.0, 6.0, 6.0, 0.0, 3.0, 6.0, 0.0, 3.0, 6.0};
    int ie[12] = {7, 7, 3, 3, 3, 3, 1, 1, 10, 10, 4, 4};
    int je[12] = {1, 3, 9, 5, 6, 4, 4, 2, 4, 11, 11, 12};
    int me[12] = {3, 8, 8, 9, 5, 6, 3, 4, 2, 4, 12, 6};
    /*边界条件*/
    float d[6] = {0.0, 0.0, 0.0, 100.0, 100.0, 100.0};
    float k[12][12] = {0};
    float b[12] = {0};
    float c[12] = {0};
    float sideta = 18;
    float ff[3] = {0};
    int t, f1, f2;
    int p, q;

    /*单元分析*/
    for (t = 0; t < 12; t++){
        b[ie[t] - 1] = y[je[t] - 1] - y[me[t] - 1];
        b[je[t] - 1] = y[me[t] - 1] - y[ie[t] - 1];
        b[me[t] - 1] = y[ie[t] - 1] - y[je[t] - 1];
        c[ie[t] - 1] = x[me[t] - 1] - x[je[t] - 1];
        c[je[t] - 1] = x[ie[t] - 1] - x[me[t] - 1];
        c[me[t] - 1] = x[je[t] - 1] - x[ie[t] - 1];
    
        ff[0] = ie[t] - 1;
        ff[1] = je[t] - 1;
        ff[2] = me[t] - 1;
    
        for (f1 = 0; f1 < 3; f1++){
            p = ff[f1];
            for (f2 = f1; f2 < 3; f2++){
                q = ff[f2];
                k[p][q] = k[p][q] + (b[p] * b[q] + c[p] * c[q]) / sideta;
                k[q][p] = k[p][q];
            }
        }
    }
    
    LS(k, d);
    
    system("pause");
    return 0;

}

void LS(float K[12][12], float d[6])
{
    int i, j;
    float temp;
    float R[6] = {0}, A[6] = {0};
    float *matrix[7];

    for (i = 0; i < 6; i++){
        matrix[i]= K[i];
    }
    
    for (i = 0; i < 6; i++){
        for (j = 6; j < 12; j++){
            R[i] += -K[i][j] * d[j - 6];
        }
    }
    matrix[6] = &R[0];
    /*高斯消元法*/
    gauss_elimination(matrix);
    
    A[5] = R[5] / (*(matrix[5] + 5));
    for (i = 1; i < 6; i++){
        temp = 0;
        for(j = 0; j < i; j++){
            temp += *(matrix[5 - j] + 5 - i) * A[5 - j];
        }
        A[5-i] = (R[5-i]-temp) / *(matrix[5-i]+5-i);
    }
    
    for (i = 0; i < 6; i++){
        printf("%-8.3f", A[i]);
    }
    putchar('\n');
    
    return;

}

void gauss_elimination(float *matrix[7])
{
    int i, j, k;
    float coe;

    for(i=0;i<6;i++){
        exchange(matrix, i);
        for(j=i+1;j<6;j++){
            coe = *(matrix[i]+j) / *(matrix[i]+i);
            for(k=i;k<7;k++){
                *(matrix[k]+j) -= coe * *(matrix[k]+i);
            }           
        }
    }
    
    return;

}

void exchange(float *matrix[7], int flag)
{
    int i;
    float*temp;

    for(i=flag+1;i<6;i++){
        if(fabs(*(matrix[flag]+flag))<fabs(*(matrix[flag]+i))){
            temp = matrix[flag];
            matrix[flag] = matrix[i];
            matrix[i] = temp;
        }
    }
    
    return;

}
